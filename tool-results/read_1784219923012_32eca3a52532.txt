    68→  const [snake, setSnake] = useState([{ x: 5, y: 5 }])

    92→        head.y += direction.y
    93→        
    94→        if (head.x < 0 || head.x >= 15 || head.y < 0 || head.y >= 15 ||

   185→// Tic Tac Toe Component
   186→function TicTacToe({ onClose }: { onClose: () => void }) {
   187→  const [board, setBoard] = useState(Array(9).fill(null))
   188→  const [isXNext, setIsXNext] = useState(true)
   189→  const [winner, setWinner] = useState<string | null>(null)
   190→
   191→  const checkWinner = (s: (string | null)[]) => {
   192→    const lines = [[0,1,2],[3,4,5],[6,7,8],[0,3,6],[1,4,7],[2,5,8],[0,4,8],[2,4,6]]
   193→    for (const [a,b,c] of lines) {
   194→      if (s[a] && s[a] === s[b] && s[a] === s[c]) return s[a]
   195→    }
   196→    return null
   197→  }
   198→
   199→  const handleClick = (i: number) => {
   200→    if (board[i] || winner) return
   201→    const newBoard = [...board]
   202→    newBoard[i] = isXNext ? '❌' : '⭕'
   203→    setBoard(newBoard)
   204→    setIsXNext(!isXNext)
   205→    const w = checkWinner(newBoard)
   206→    if (w) setWinner(w)
   207→  }
   208→
   209→  const reset = () => { setBoard(Array(9).fill(null)); setIsXNext(true); setWinner(null) }
   210→
   211→  return (
   212→    <div style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', gap: '12px', height: '100%', justifyContent: 'center' }}>
   213→      <div style={{ display: 'flex', justifyContent: 'space-between', width: '100%' }}>
   214→        <span style={{ color: '#ef4444', fontWeight: 'bold' }}>{winner ? `${winner} wins!` : isXNext ? "❌'s Turn" : "⭕'s Turn"}</span>
   215→        <button onClick={onClose} style={closeBtnStyle}>✕</button>
   216→      </div>
   217→      <div style={{ display: 'grid', gridTemplateColumns: 'repeat(3, 60px)', gap: '8px' }}>
   218→        {board.map((cell, i) => (
   219→          <button key={i} onClick={() => handleClick(i)} style={{
   220→            width: 60, height: 60, fontSize: '28px',
   221→            background: cell ? 'rgba(255,255,255,0.1)' : 'rgba(255,255,255,0.05)',
   222→            border: '1px solid rgba(255,255,255,0.2)', borderRadius: '12px',
   223→            color: '#fff', cursor: 'pointer', transition: 'all 0.15s'
   224→          }}>{cell}</button>
   225→        ))}
   226→      </div>
   227→      <button onClick={reset} style={{ ...appBtnStyle, background: 'linear-gradient(135deg, #ef4444, #dc2626)' }}>New Game</button>
   228→    </div>
   229→  )
   230→}

   232→// Memory Match Game
   233→function MemoryGame({ onClose }: { onClose: () => void }) {
   234→  const emojis = ['🎮', '🎲', '🎯', '🎪', '🎨', '🎭', '🎪', '🎯']
   235→  const [cards, setCards] = useState<{ emoji: string; flipped: boolean; matched: boolean }[]>(() =>
   236→    [...emojis, ...emojis].sort(() => Math.random() - 0.5).map(e => ({ emoji: e, flipped: false, matched: false }))
   237→  )
   238→  const [flippedIdx, setFlippedIdx] = useState<number[]>([])
   239→  const [moves, setMoves] = useState(0)
   240→
   241→  const handleClick = (idx: number) => {
   242→    if (flippedIdx.length >= 2 || cards[idx].flipped || cards[idx].matched) return
   243→    const newCards = [...cards]
   244→    newCards[idx].flipped = true
   245→    setCards(newCards)
   246→    const newFlipped = [...flippedIdx, idx]
   247→    setFlippedIdx(newFlipped)
   248→
   249→    if (newFlipped.length === 2) {
   250→      setMoves(m => m + 1)
   251→      if (cards[newFlipped[0]].emoji === cards[newFlipped[1]].emoji) {
   252→        setTimeout(() => {
   253→          const c = [...cards]
   254→          c[newFlipped[0]].matched = true
   255→          c[newFlipped[1]].matched = true
   256→          setCards(c)
   257→          setFlippedIdx([])
   258→        }, 500)
   259→      } else {
   260→        setTimeout(() => {
   261→          const c = [...cards]
   262→          c[newFlipped[0]].flipped = false
   263→          c[newFlipped[1]].flipped = false
   264→          setCards(c)
   265→          setFlippedIdx([])
   266→        }, 800)
   267→      }
   268→    }
   269→  }
   270→
   271→  const reset = () => {
   272→    setCards([...emojis, ...emojis].sort(() => Math.random() - 0.5).map(e => ({ emoji: e, flipped: false, matched: false })))
   273→    setFlippedIdx([])
   274→    setMoves(0)
   275→  }
   276→
   277→  const won = cards.every(c => c.matched)
   278→
   279→  return (
   280→    <div style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', gap: '10px', height: '100%' }}>
   281→      <div style={{ display: 'flex', justifyContent: 'space-between', width: '100%' }}>
   282→        <span style={{ color: '#8b5cf6', fontWeight: 'bold' }}>Moves: {moves}</span>
   283→        <button onClick={onClose} style={closeBtnStyle}>✕</button>
   284→      </div>
   285→      <div style={{ display: 'grid', gridTemplateColumns: 'repeat(4, 52px)', gap: '8px' }}>
   286→        {cards.map((card, i) => (
   287→          <button key={i} onClick={() => handleClick(i)} style={{
   288→            width: 52, height: 52, fontSize: '24px',
   289→            background: card.flipped || card.matched ? 'rgba(139,92,246,0.2)' : 'linear-gradient(135deg, #8b5cf6, #7c3aed)',
   290→            border: card.matched ? '2px solid #22c55e' : '1px solid rgba(255,255,255,0.2)',
   291→            borderRadius: '10px', color: card.flipped || card.matched ? '#fff' : 'transparent',
   292→            cursor: 'pointer', transition: 'all 0.2s'
   293→          }}>{card.flipped || card.matched ? card.emoji : '?'}</button>
   294→        ))}
   295→      </div>
   296→      {won && <div style={{ color: '#22c55e', fontWeight: 'bold' }}>🎉 You Won!</div>}
   297→      <button onClick={reset} style={{ ...appBtnStyle, background: 'linear-gradient(135deg, #8b5cf6, #7c3aed)' }}>Reset</button>
   298→    </div>
   299→  )
   300→}

   302→// 2048 Puzzle Game
   303→function Puzzle2048({ onClose }: { onClose: () => void }) {
   304→  const [grid, setGrid] = useState<number[][]>(() => Array(4).fill(null).map(() => Array(4).fill(0)))
   305→  const [score, setScore] = useState(0)
   306→
   307→  const addTile = (g: number[][]) => {
   308→    const empty: [number, number][] = []
   309→    g.forEach((row, r) => row.forEach((cell, c) => { if (cell === 0) empty.push([r, c]) }))
   310→    if (empty.length > 0) {
   311→      const [r, c] = empty[Math.floor(Math.random() * empty.length)]
   312→      g[r][c] = Math.random() < 0.9 ? 2 : 4
   313→    }
   314→    return g
   315→  }
   316→
   317→  const initGame = () => {
   318→    let g = Array(4).fill(null).map(() => Array(4).fill(0))
   319→    g = addTile(g)
   320→    g = addTile(g)
   321→    setGrid(g)
   322→    setScore(0)
   323→  }
   324→
   325→  useEffect(() => { initGame() }, [])
   326→
   327→  const slide = (g: number[][], dir: string): number[][] => {
   328→    let newG = g.map(row => [...row])
   329→    const rotated = dir === 'up' || dir === 'down' ? newG[0].map((_, i) => newG.map(row => row[i])) : newG.map(row => [...row])
   330→    
   331→    const moved = rotated.map(row => {
   332→      let filtered = row.filter(x => x !== 0)
   333→      for (let i = 0; i < filtered.length - 1; i++) {
   334→        if (filtered[i] === filtered[i + 1]) {
   335→          filtered[i] *= 2
   336→          setScore(s => s + filtered[i])
   337→          filtered.splice(i + 1, 1)
   338→        }
   339→      }
   340→      while (filtered.length < 4) filtered.push(0)
   341→      return filtered
   342→    })
   343→
   344→    return dir === 'up' || dir === 'down' ? moved[0].map((_, i) => moved.map(row => row[i])) : moved
   345→  }
   346→
   347→  const move = (dir: string) => {
   348→    const newGrid = slide(grid, dir)
   349→    if (JSON.stringify(newGrid) !== JSON.stringify(grid)) {
   350→      setGrid(addTile(newGrid))
   351→    }
   352→  }
   353→
   354→  useEffect(() => {
   355→    const handleKey = (e: KeyboardEvent) => {
   356→      switch(e.key) {
   357→        case 'ArrowUp': move('up'); break
   358→        case 'ArrowDown': move('down'); break
   359→        case 'ArrowLeft': move('left'); break
   360→        case 'ArrowRight': move('right'); break
   361→      }
   362→    }
   363→    window.addEventListener('keydown', handleKey)
   364→    return () => window.removeEventListener('keydown', handleKey)
   365→  }, [grid])
   366→
   367→  const getColor = (val: number) => {
   368→    const colors: Record<number, string> = { 2: '#eee4da', 4: '#ede0c8', 8: '#f2b179', 16: '#f59563', 32: '#f67c5f', 64: '#f65e3b', 128: '#edcf72', 256: '#edcc61', 512: '#edc850', 1024: '#edc53f', 2048: '#edc22e' }
   369→    return colors[val] || '#3c3a32'
   370→  }
   371→
   372→  return (
   373→    <div style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', gap: '10px', height: '100%' }}>
   374→      <div style={{ display: 'flex', justifyContent: 'space-between', width: '100%' }}>
   375→        <span style={{ color: '#f59e0b', fontWeight: 'bold' }}>Score: {score}</span>
   376→        <button onClick={onClose} style={closeBtnStyle}>✕</button>
   377→      </div>
   378→      <div style={{ display: 'grid', gridTemplateColumns: 'repeat(4, 52px)', gap: '6px', background: '#bbada0', padding: '8px', borderRadius: '8px' }}>
   379→        {grid.flat().map((cell, i) => (
   380→          <div key={i} style={{
   381→            width: 52, height: 52, display: 'flex', alignItems: 'center', justifyContent: 'center',
   382→            background: getColor(cell), borderRadius: '6px', fontSize: cell > 512 ? '18px' : cell > 64 ? '22px' : '26px',
   383→            fontWeight: 'bold', color: cell > 4 ? '#f9f6f2' : '#776e65'
   384→          }}>{cell || ''}</div>
   385→        ))}
   386→      </div>
   387→      <div style={{ display: 'flex', gap: '8px' }}>
   388→        <button onClick={() => move('left')} style={arrowBtnStyle}>←</button>
   389→        <button onClick={() => move('up')} style={arrowBtnStyle}>↑</button>
   390→        <button onClick={() => move('down')} style={arrowBtnStyle}>↓</button>
   391→        <button onClick={() => move('right')} style={arrowBtnStyle}>→</button>
   392→      </div>
   393→      <button onClick={initGame} style={{ ...appBtnStyle, background: 'linear-gradient(135deg, #f59e0b, #d97706)' }}>New Game</button>
   394→    </div>
   395→  )
   396→}

   462→function CalculatorApp({ onClose }: { onClose: () => void }) {
   463→  const [display, setDisplay] = useState('0')
   464→  const [prevValue, setPrevValue] = useState<number | null>(null)
   465→  const [operator, setOperator] = useState<string | null>(null)
   466→  const [waitingForOperand, setWaitingForOperand] = useState(false)
   467→
   468→  const inputDigit = (digit: string) => {
   469→    if (waitingForOperand) { setDisplay(digit); setWaitingForOperand(false) }
   470→    else setDisplay(display === '0' ? digit : display + digit)
   471→  }
   472→
   473→  const inputDot = () => {
   474→    if (waitingForOperand) { setDisplay('0.'); setWaitingForOperand(false); return }
   475→    if (!display.includes('.')) setDisplay(display + '.')
   476→  }
   477→
   478→  const clear = () => { setDisplay('0'); setPrevValue(null); setOperator(null) }
   479→
   480→  const performOperation = (nextOp: string) => {
   481→    const inputValue = parseFloat(display)
   482→    if (prevValue === null) { setPrevValue(inputValue) }
   483→    else if (operator) {
   484→      const result = calculate(prevValue, inputValue, operator)
   485→      setDisplay(String(result))
   486→      setPrevValue(result)
   487→    }
   488→    setWaitingForOperand(true)
   489→    setOperator(nextOp)
   490→  }
   491→
   492→  const calculate = (a: number, b: number, op: string): number => {
   493→    switch(op) { case '+': return a + b; case '-': return a - b; case '*': return a * b; case '/': return b !== 0 ? a / b : 0; default: return b }
   494→  }
   495→
   496→  const equals = () => {
   497→    if (!operator || prevValue === null) return
   498→    const result = calculate(prevValue, parseFloat(display), operator)
   499→    setDisplay(String(result))
   500→    setPrevValue(null)
   501→    setOperator(null)
   502→    setWaitingForOperand(true)
   503→  }
   504→
   505→  const btnStyle = (color: string) => ({
   506→    width: '50px', height: '50px', borderRadius: '12px', border: 'none',
   507→    background: color, color: '#fff', fontSize: '20px', fontWeight: '600',
   508→    cursor: 'pointer', display: 'flex', alignItems: 'center', justifyContent: 'center',
   509→    transition: 'transform 0.1s', userSelect: 'none'
   510→  })
   511→
   512→  return (
   513→    <div style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', gap: '10px', height: '100%' }}>
   514→      <div style={{ display: 'flex', justifyContent: 'space-between', width: '100%' }}>
   515→        <span />
   516→        <button onClick={onClose} style={closeBtnStyle}>✕</button>
   517→      </div>
   518→      <div style={{
   519→        width: '100%', background: '#1e1e2e', borderRadius: '12px', padding: '16px',
   520→        textAlign: 'right', fontSize: '32px', fontWeight: '300', color: '#fff',
   521→        minHeight: '50px', wordBreak: 'break-all'
   522→      }}>{display}</div>
   523→      <div style={{ display: 'grid', gridTemplateColumns: 'repeat(4, 50px)', gap: '8px' }}>
   524→        <button onClick={clear} style={btnStyle('#475569')}>C</button>
   525→        <button onClick={() => setDisplay(display.startsWith('-') ? display.slice(1) : '-' + display)} style={btnStyle('#475569')}>±</button>
   526→        <button onClick={() => setDisplay(String(parseFloat(display) / 100))} style={btnStyle('#475569')}>%</button>
   527→        <button onClick={() => performOperation('/')} style={btnStyle('#f59e0b')}>÷</button>
   528→        {['7','8','9'].map(n => <button key={n} onClick={() => inputDigit(n)} style={btnStyle('#334155')}>{n}</button>)}
   529→        <button onClick={() => performOperation('*')} style={btnStyle('#f59e0b')}>×</button>
   530→        {['4','5','6'].map(n => <button key={n} onClick={() => inputDigit(n)} style={btnStyle('#334155')}>{n}</button>)}
   531→        <button onClick={() => performOperation('-')} style={btnStyle('#f59e0b')}>−</button>
   532→        {['1','2','3'].map(n => <button key={n} onClick={() => inputDigit(n)} style={btnStyle('#334155')}>{n}</button>)}
   533→        <button onClick={() => performOperation('+')} style={btnStyle('#f59e0b')}>+</button>
   534→        <button onClick={() => inputDigit('0')} style={{ ...btnStyle('#334155'), gridColumn: 'span 2', width: '108px' }}>0</button>
   535→        <button onClick={inputDot} style={btnStyle('#334155')}>.</button>
   536→        <button onClick={equals} style={btnStyle('#22c55e')}>=</button>
   537→      </div>
   538→    </div>
   539→  )

   630→// Paint App
   631→function PaintApp({ onClose }: { onClose: () => void }) {
   632→  const canvasRef = useRef<HTMLCanvasElement>(null)
   633→  const [drawing, setDrawing] = useState(false)
   634→  const [color, setColor] = useState('#ffffff')
   635→  const [brushSize, setBrushSize] = useState(4)
   636→  const lastPos = useRef<{ x: number; y: number }>({ x: 0, y: 0 })
   637→
   638→  useEffect(() => {
   639→    const canvas = canvasRef.current
   640→    if (canvas) {
   641→      canvas.width = canvas.offsetWidth
   642→      canvas.height = canvas.offsetHeight
   643→      const ctx = canvas.getContext('2d')
   644→      if (ctx) ctx.fillStyle = '#1a1a2e'; ctx.fillRect(0, 0, canvas.width, canvas.height)
   645→    }
   646→  }, [])
   647→
   648→  const getPos = (e: React.TouchEvent | React.MouseEvent) => {
   649→    const canvas = canvasRef.current!
   650→    const rect = canvas.getBoundingClientRect()
   651→    if ('touches' in e) return { x: e.touches[0].clientX - rect.left, y: e.touches[0].clientY - rect.top }
   652→    return { x: e.clientX - rect.left, y: e.clientY - rect.top }
   653→  }
   654→
   655→  const startDraw = (e: React.TouchEvent | React.MouseEvent) => {
   656→    setDrawing(true)
   657→    lastPos.current = getPos(e)
   658→  }
   659→
   660→  const draw = (e: React.TouchEvent | React.MouseEvent) => {
   661→    if (!drawing) return
   662→    const canvas = canvasRef.current!
   663→    const ctx = canvas.getContext('2d')!
   664→    const pos = getPos(e)
   665→    ctx.strokeStyle = color; ctx.lineWidth = brushSize; ctx.lineCap = 'round'
   666→    ctx.beginPath(); ctx.moveTo(lastPos.current.x, lastPos.current.y); ctx.lineTo(pos.x, pos.y); ctx.stroke()
   667→    lastPos.current = pos
   668→  }
   669→
   670→  const stopDraw = () => setDrawing(false)
   671→  const clearCanvas = () => {
   672→    const canvas = canvasRef.current!
   673→    const ctx = canvas.getContext('2d')!
   674→    ctx.fillStyle = '#1a1a2e'; ctx.fillRect(0, 0, canvas.width, canvas.height)
   675→  }
   676→
   677→  const colors = ['#ffffff', '#ef4444', '#f59e0b', '#22c55e', '#3b82f6', '#8b5cf6', '#ec4899']
   678→
   679→  return (
   680→    <div style={{ display: 'flex', flexDirection: 'column', height: '100%', gap: '8px' }}>
   681→      <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center' }}>
   682→        <span style={{ color: '#ec4899', fontWeight: 'bold' }}>🎨 Paint</span>
   683→        <button onClick={onClose} style={closeBtnStyle}>✕</button>
   684→      </div>
   685→      <canvas ref={canvasRef}
   686→        onMouseDown={startDraw} onMouseMove={draw} onMouseUp={stopDraw} onMouseLeave={stopDraw}
   687→        onTouchStart={startDraw} onTouchMove={draw} onTouchEnd={stopDraw}
   688→        style={{ flex: 1, borderRadius: '12px', touchAction: 'none', cursor: 'crosshair' }}
   689→      />
   690→      <div style={{ display: 'flex', gap: '6px', alignItems: 'center', flexWrap: 'wrap' }}>
   691→        {colors.map(c => (
   692→          <button key={c} onClick={() => setColor(c)} style={{
   693→            width: 28, height: 28, borderRadius: '50%', background: c,
   694→            border: color === c ? '3px #fff solid' : '2px rgba(255,255,255,0.2) solid', cursor: 'pointer'
   695→          }} />
   696→        ))}
   697→        <input type="range" min="1" max="20" value={brushSize} onChange={e => setBrushSize(+e.target.value)} style={{ width: '80px', accentColor: '#ec4899' }} />
   698→        <button onClick={clearCanvas} style={{ ...appBtnStyle, background: '#64748b', padding: '6px 12px', fontSize: '11px' }}>Clear</button>
   699→      </div>
   700→    </div>
   701→  )
   702→}

  1034→// Age Calculator App
  1035→function AgeCalcApp({ onClose }: { onClose: () => void }) {
  1036→  const [birthDate, setBirthDate] = useState('')
  1037→  const [result, setResult] = useState<{ years: number; months: number; days: number; nextBirthday: number } | null>(null)
  1038→
  1039→  const calculateAge = () => {
  1040→    if (!birthDate) return
  1041→    const birth = new Date(birthDate)
  1042→    const today = new Date()
  1043→    let years = today.getFullYear() - birth.getFullYear()
  1044→    let months = today.getMonth() - birth.getMonth()
  1045→    let days = today.getDate() - birth.getDate()
  1046→    
  1047→    if (days < 0) { months--; days += new Date(today.getFullYear(), today.getMonth(), 0).getDate() }
  1048→    if (months < 0) { years--; months += 12 }
  1049→    
  1050→    const nextBirth = new Date(today.getFullYear(), birth.getMonth(), birth.getDate())
  1051→    if (nextBirth < today) nextBirth.setFullYear(nextBirth.getFullYear() + 1)
  1052→    const daysUntil = Math.ceil((nextBirth.getTime() - today.getTime()) / (1000*60*60*24))
  1053→    
  1054→    setResult({ years, months, days, nextBirthday: daysUntil })
  1055→  }
  1056→
  1057→  return (
  1058→    <div style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', gap: '16px', height: '100%', justifyContent: 'center', padding: '16px' }}>
  1059→      <div style={{ display: 'flex', justifyContent: 'space-between', width: '100%' }}>
  1060→        <span style={{ color: '#fb923c', fontWeight: 'bold' }}>🎂 Age Calculator</span>
  1061→        <button onClick={onClose} style={closeBtnStyle}>✕</button>
  1062→      </div>
  1063→      <input type="date" value={birthDate} onChange={e => setBirthDate(e.target.value)} max={new Date().toISOString().split('T')[0]} style={{
  1064→        background: '#1e1e2e', border: '1px solid rgba(255,255,255,0.2)', borderRadius: '12px',
  1065→        padding: '14px 18px', color: '#fff', fontSize: '16px', outline: 'none', width: '100%'
  1066→      }} />
  1067→      <button onClick={calculateAge} style={{ ...appBtnStyle, background: 'linear-gradient(135deg, #fb923c, #f97316)', width: '100%' }}>
  1068→        Calculate My Age! 🎉
  1069→      </button>
  1070→      {result && (
  1071→        <div style={{
  1072→          background: 'rgba(251,146,60,0.1)', border: '1px solid rgba(251,146,60,0.3)',
  1073→          borderRadius: '16px', padding: '20px', width: '100%', textAlign: 'center'
